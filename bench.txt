#!/usr/bin/env python3
# Compat: permet d'executer le micro-benchmark via `python bench.txt`.
"""Compat: permet d'executer le micro-benchmark via `python bench.txt`."""

from __future__ import annotations

from pathlib import Path


def _sanity_check_source(path: Path) -> None:
    src = path.read_text(encoding="utf-8")
    markers = ("<<<<<<<", "=======", ">>>>>>>")
    repaired = False
    if any(m in src for m in markers):
        # Tentative de réparation minimale: suppression des lignes de marqueurs.
        src = "\n".join(
            line for line in src.splitlines()
            if not line.startswith(("<<<<<<<", "=======", ">>>>>>>"))
        ) + "\n"
        repaired = True

    # Certains merges Windows ont laissé un docstring `verify` invalide.
    bad_verify = '"""Retourne (amicale?, hard?).  hard = m non factorise avec la borne."""'
    if bad_verify in src:
        src = src.replace(
            bad_verify,
            "# Retourne (amicale?, hard?) ; hard = m non factorise avec la borne.",
        )
        repaired = True

    if repaired:
        path.write_text(src, encoding="utf-8")

    if any(m in src for m in markers):
        raise RuntimeError(
            f"Conflit de merge detecte dans {path.name} "
            "(marqueurs <<<<<<< / ======= / >>>>>>>)."
        )
    # Vérifie aussi qu'il n'y a pas d'erreur de parsing Python.
    try:
        compile(src, str(path), "exec")
    except SyntaxError as exc:
        raise RuntimeError(
            f"SyntaxError dans {path.name}:{exc.lineno}. "
            "Le fichier semble contenir un merge incomplet. "
            "Supprime les marqueurs de conflit et garde une seule version "
            "de chaque fonction (current/incoming/both nettoye)."
        ) from exc
    compile(src, str(path), "exec")


if __name__ == "__main__":
    base = Path(__file__).resolve().parent
    _sanity_check_source(base / "even_odd_amicable_v2.py")
    _sanity_check_source(base / "micro_benchmark.py")
    _sanity_check_source(base / "even_odd_amicable_gmp.py")
    from micro_benchmark import main
from micro_benchmark import main


if __name__ == "__main__":
    main()
