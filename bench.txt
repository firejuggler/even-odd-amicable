#!/usr/bin/env python3
"""Compat: permet d'executer le micro-benchmark via `python bench.txt`."""

from __future__ import annotations

from pathlib import Path


def _sanity_check_source(path: Path) -> None:
    src = path.read_text(encoding="utf-8")
    markers = ("<<<<<<<", "=======", ">>>>>>>")
    if any(m in src for m in markers):
        raise RuntimeError(
            f"Conflit de merge detecte dans {path.name} "
            "(marqueurs <<<<<<< / ======= / >>>>>>>)."
        )
    # Vérifie aussi qu'il n'y a pas d'erreur de parsing Python.
    compile(src, str(path), "exec")


if __name__ == "__main__":
    _sanity_check_source(Path(__file__).with_name("even_odd_amicable_v2.py"))
    from micro_benchmark import main
    main()
